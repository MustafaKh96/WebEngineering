    function f_person(vorname, nachname, alter, auto){ // Auto MUSS als Array übergeben werden. z.B: [auto1] oder [auto1, auto2].
        this.vorname = vorname;
        this.nachname = nachname;
        this.alter = alter;
        if(auto == null){
            this.auto = auto;
        }
        else{
          auto.anzahlDerBesitzer = auto.anzahlDerBesitzer + 1;
          this.auto = auto;
        }
        this.addAuto = function(new_auto){ // Hier kann man einer Person ein Auto hinzufügen :)
            let erg = "";
            if(auto.includes(new_auto)){
                erg = "Auto exsistiert schon";
            }
            else{
                new_auto.anzahlDerBesitzer = new_auto.anzahlDerBesitzer + 1;
                this.auto.push(new_auto);
                erg = "Auto ist geaddet";
            }
            return erg;
        }

        this.welcheAutos = function(){ // Hier kann man sehen, welche Autos man hat.
            let erg = this.vorname +" hat ";
            if(auto == null){
                erg += " kein Auto.";
                
            }
            else{
              for (let i = 0; i < auto.length; i++) {
                if(i == auto.length-1){
                  erg += auto[i].type + ".";
                }
                else{
                  erg += auto[i].type + " und ";
                }
              }
            }
            return erg;	  
        }
      }
