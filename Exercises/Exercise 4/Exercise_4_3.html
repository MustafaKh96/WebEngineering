Achten Sie auf Performanz: Berechnen Sie jeden Fibonacci-Wert nur einmal. Speichern Sie zu diesem Zweck jede bereits berechnete Fibonacci-Zahl in einer Tabelle. 

        const anzahl = parseInt(prompt('Geben Sie die Anzahl der Fibonacci-Folge '));
        let erg = "";
        let zahl1 = 0;
        let zahl2 = 1;
        let dieNaechste;
        for (let i = 1; i <= anzahl; i++) {
            erg+= zahl1 + ",";
            dieNaechste = zahl1 + zahl2;
            zahl1 = zahl2;
            zahl2 = dieNaechste;
        }
        erg = erg.substring(0, erg.length -1); 
        console.log(erg);
