11.2 Menü-Komponente

11.1 Komponente in Vue.js 11.2 Menü-Komponente 11.3 Vue.js WWW-Navigator

Schreiben Sie eine möglichst flexible Vue.js Single File Component für Menüs und wenden Sie diese in derselben Webseite zweimal an, einmal horizontal, das andere Mal vertikal. 

    --------index.html--------

    <script src="https://unpkg.com/vue"></script>
    <div id="app">
        <single-file-component></single-file-component>
        <counter></counter>
    </div>
    <script type="module" src="./app.mjs"></script>

    --------app.mjs--------

    import SingleFileComponent from './SingleFileComponent.mjs';
    import Counter from './Counter.mjs';

    new Vue({
        el: '#app',
        components: {
        SingleFileComponent,
        Counter
        }
        });

    --------SingleFileComponent.mjs--------

    export default{
        template:`<ul>
                    <li>Option1</li>
                    <li>Option2</li>
                    <li>Option3</li>
                  </ul>`
    }

    --------Counter.mjs--------

    export default{
        template:`<ul>
                    <li :style="{'float': 'left', 'margin-right': '15px', 'list-style': 'none' }">Option1</li>
                    <li :style="{'float': 'left', 'margin-right': '15px', 'list-style': 'none' }">Option2</li>
                    <li :style="{'float': 'left', 'margin-right': '15px', 'list-style': 'none' }">Option3</li>
                  </ul>`
    }
